Usage with React

import React, { useRef, useEffect } from 'react';
import '@semantic-protocol/web-components';

function App() {
  const fieldRef = useRef(null);
  
  useEffect(() => {
    // Access web component methods
    if (fieldRef.current) {
      fieldRef.current.addEventListener('semantic-change', (e) => {
        console.log('Value changed:', e.detail.value);
      });
    }
  }, []);
  
  return (
    <semantic-provider>
      <semantic-field
        ref={fieldRef}
        type="email"
        name="email"
        label="Email Address"
        required
      />
      
      <semantic-discovery
        type="input"
        display-mode="grid"
      />
    </semantic-provider>
  );
}

// Or use the React wrapper
import { createFrameworkWrapper } from '@semantic-protocol/web-components';
import { SemanticField } from '@semantic-protocol/web-components';

const SemanticFieldReact = createFrameworkWrapper(SemanticField, 'react');

function FormComponent() {
  const [email, setEmail] = useState('');
  
  return (
    <SemanticFieldReact
      type="email"
      value={email}
      onChange={(e) => setEmail(e.detail.value)}
      onSemanticValidate={(e) => console.log('Valid:', e.detail.valid)}
    />
  );
}

Usage with Vue

// main.js - Register globally
import { defineElements } from '@semantic-protocol/web-components';
defineElements();

// Component.vue
<template>
  <semantic-provider>
    <semantic-field
      v-model="email"
      type="email"
      name="email"
      label="Email Address"
      :required="true"
      @semantic-change="handleChange"
    />
    
    <semantic-discovery
      :query="discoveryQuery"
      display-mode="list"
    />
  </semantic-provider>
</template>

<script setup>
import { ref, computed } from 'vue';

const email = ref('');
const discoveryQuery = computed(() => ({
  type: 'input',
  flow: 'registration'
}));

const handleChange = (event) => {
  console.log('Field changed:', event.detail);
};
</script>

// Or use Vue wrapper
import { createFrameworkWrapper } from '@semantic-protocol/web-components';
import { SemanticField } from '@semantic-protocol/web-components';

export default {
  components: {
    SemanticField: createFrameworkWrapper(SemanticField, 'vue')
  }
};

Angular Integration Angular

Module Setup

// app.module.ts
import { NgModule, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
import { defineElements } from '@semantic-protocol/web-components';

// Define custom elements
defineElements();

@NgModule({
  schemas: [CUSTOM_ELEMENTS_SCHEMA]
})
export class AppModule {}

Component Usage

// app.component.ts
import { Component, ViewChild, ElementRef, AfterViewInit } from '@angular/core';

@Component({
  selector: 'app-root',
  template: `
    <semantic-provider>
      <semantic-field
        #emailField
        type="email"
        name="email"
        label="Email Address"
        [attr.required]="true"
        (semantic-change)="onEmailChange($event)"
      ></semantic-field>
      
      <semantic-discovery
        [attr.type]="'input'"
        [attr.display-mode]="'grid'"
      ></semantic-discovery>
    </semantic-provider>
  `
})
export class AppComponent implements AfterViewInit {
  @ViewChild('emailField') emailField!: ElementRef;
  
  ngAfterViewInit() {
    // Access web component methods
    const field = this.emailField.nativeElement;
    field.validate();
  }
  
  onEmailChange(event: CustomEvent) {
    console.log('Email changed:', event.detail.value);
  }
}

Svelte Integration Svelte

Component Usage

<script>
  import { onMount } from 'svelte';
  import '@semantic-protocol/web-components';
  
  let email = '';
  let fieldElement;
  
  onMount(() => {
    // Custom elements are automatically available
    if (fieldElement) {
      fieldElement.addEventListener('semantic-change', (e) => {
        email = e.detail.value;
      });
    }
  });
  
  function handleSubmit() {
    if (fieldElement.validate()) {
      console.log('Valid email:', email);
    }
  }
</script>

<semantic-provider>
  <semantic-field
    bind:this={fieldElement}
    type="email"
    name="email"
    label="Email Address"
    required
  />
  
  <button on:click={handleSubmit}>Submit</button>
  
  <semantic-discovery
    type="input"
    display-mode="list"
  />
</semantic-provider>

CDN Usage

<!-- Auto-initializing version -->
<script src="https://unpkg.com/@semantic-protocol/web-components/dist/semantic-components.auto.js"></script>

<!-- Manual initialization -->
<script src="https://unpkg.com/@semantic-protocol/web-components/dist/semantic-components.umd.js"></script>
<script>
  // Initialize components
  SemanticComponents.defineElements();
  
  // Use the components
  document.addEventListener('DOMContentLoaded', () => {
    const provider = document.createElement('semantic-provider');
    provider.setAttribute('debug', 'true');
    
    const field = document.createElement('semantic-field');
    field.type = 'email';
    field.name = 'email';
    field.label = 'Email Address';
    field.required = true;
    
    // Listen for events
    field.addEventListener('semantic-change', (e) => {
      console.log('Value:', e.detail.value);
    });
    
    // Validate programmatically
    field.validate();
    
    // Get value
    const value = field.value;
    
    provider.appendChild(field);
    document.body.appendChild(provider);
  });
</script>
